BEGINSUB Name=GcApi2 
# Usage:
# Check for errors when making API calls. Just call once before a loop and then, inside the loop, check for specifics, such as speeding tickets.

# Parameters: 
# $url (For API call)
# $verb (Defaults to "Get") Methods can be Get, Post, Put or Delete
# $json (Defaults empty string) Formatted Json data. 
#    Only meaningful (and required) for Post and Put methods
# $apiCaption (Defaults empty string) for dialog feedback

# Returned Variables:
#	$apiError=True if there is an error
# $data=Result in JSON format. Usually not used. The ApiMaster or GcApi2 tables are accessed for resulting data.
# $msg=Report of error information. A YesNo dialog will be displayed to allow the macro to continue or cancel. It is also copied to the Clipboard.

# Calling Code:
	# $url="The API request"
	# $apiCaption="Feedback Caption for the API call"
	# GOSUB Name=GcApi2 # Call this sub
	# IF Not($apiError) # OR # IF $apiError # OR # use ELSE
		# Code to process returned API data
	# ENDIF 
	# IF $apiNoData
		 # skip / continue / fallback
	# ENDIF

# Notes:
# If a valid API request returns no data, $apiNoData=True.
# This is NOT treated as an API error and no dialog is shown.
# An informational message is stored in $msg.
# The calling code is responsible for handling this condition, e.g.:
#
# IF $apiNoData
#   display or handle $msg as required
# ENDIF

# Clean start
	$apiError=False
	$apiNoData=False
	$apiLog="$_AppData\apilog.txt"
	$statusCode=0

	IF $verb="" # Only True on first call in macro
		$verb="Get"
	ENDIF

# Call API
  $data=GcApi2($url, $verb, $json, $apiCaption)
	$apiTotal=Val(RegExSub("x-total-count: (\d+)",$_GcApi2Header,1,1))

	BEGINCASE
		CASE $data="[]" AND Not($_GcApiError)
			# Next most likely case is a valid API call that returns no data
			$msg="The API call has not returned any data.$p There was no API Error so the request was valid but there is no data that matches the request.$p If you believe data should have been returned, then create a 'Support Post' for this macro and paste in the message from the Clipboard.$p The API Request was:$br $url"
			$apiNoData=True
		CASE $_GcApiError
		  # We have an API error so get details
			$statusCode=Val(RegExSub("statusCode"":(\d+),",$data,1,1))
			$statusMessage=RegExSub("statusMessage"":""(.*?)"",",$data,1,1)
			$errorMessage=RegExSub("errorMessage"":""(.*?)""",$data,1,1)
			$limitReset=RegExSub("x-rate-limit-reset: (\d+)",$_GcApi2Header,1,1)
			$optedOut=RegExSub("x-opted-out: (.*)",$_GcApi2Header,1,1)
			$apiError=True
			$msg="$apiCaption$p API Request$br $url$p statusCode=$statusCode$br statusMessage=$statusMessage$br errorMessage=$errorMessage$br limitReset=$limitReset$br apiTotal=$apiTotal$br optedOut=$optedOut$br GcApiError=$_GcApiError$p If this error persists, Create a Support post for this macro and paste in the message from the Clipboard. "
			BEGINCASE
				CASE $statusCode=429 # Speeding Ticket
# api calls exceeded maximum allowed 60 per Minute, so delay and try again					
					$delay=Val(Regexsub("x-rate-limit-reset: (\d+)", $_GcApi2Header,1,1))
					SHOWSTATUS msg="Waiting $delay seconds (Exceeded 60 calls per min)" width=400 top=4 left=4 
					$delay=$delay*1000
					DELAY Ms=$delay
					SHOWSTATUS Display=Off
					GOSUB Name=GcApi2
					EXITSUB
				CASE $statusCode=401 # Unauthorized
						$msg=$msg+"Unauthorized.$p Go to GSAK -> Geocaching.com Access -> Get Another Access Token and authorize GSAK to access your account."
				CASE $statusCode=403 # Forbidden 
					$msg="Forbidden.$p Mostly likely caused by trying to access data for opted-out users or non-public Adventures.$p"+$msg
				CASE SysInfo("Internet")="no" # Fix for temporary loss of Internet and try again
					IF YesNo("You have lost Internet connection.$p Select 'Yes' once you have reconnected or 'No' to Cancel.")
						GOSUB Name=GcApi2
					ELSE
						CANCEL
					ENDIF
				CASE $statusCode=0 AND $errorMessage="" # No help here
# ApiLog is only erased if we get here
					IF $logErased
# Check for 401 Unauthorized
						$file=GetFile($APILog)
# This could become a CASE/OTHERWISE block if more errors discovered
						IF At("401 Unauthorized", $file)<>0

 # $codes: Comma-separated list of GC codes

  ENDIF # Check BlockSize

    $liteformat=True
    $fullformat=False

  IF $_Sql<>""

    $S=Sqlite("sql",$_Sql)
    $S=REPLACE($_Crlf,",",$S)
  ELSE # Assign formatted codes to $S for maipulation

    IF $N>0

    $GcGeocachesCodes=$gcGeocachesTemp
