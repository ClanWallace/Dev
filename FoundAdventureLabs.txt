BEGINSUB Name=CreateAdventureLabs
debug status=on
  $url="adventures/$advId"
  IF $advId<>""
    $apiCaption="Getting Adventure Lab Details"
    GOSUB Name=GcApi2

    IF Not($apiError)
      $isArchived=False
    ELSE
      BEGINCASE
        CASE $statusCode=403
          IF $errorMessage="Can only retrieve public adventure data."
            $isArchived=True
          ENDIF
        CASE $statusCode=404
          IF $errorMessage="Failed to get results from adventures service" OR At("Adventure with id",$errorMessage)>0
            $isArchived=True
          ENDIF
        CASE $statusCode=500
          $isArchived=True
        OTHERWISE
          $isArchived=False
      ENDCASE
    ENDIF

    $B=False
    IF $isArchived
      # Try this API to get the Deeplink
      $S=GetUrl("https://api.lab2gpx.gcutils.de/lab2gpx-data/by-uuid/$advId")
      $SS=RegExSub($_Quote+"url"+$_Quote+"\s*:\s*"+$_Quote+"(.*?)"+$_Quote,$S,1,1)
      IF $SS<>""
        $url="https://labs.geocaching.com/goto/"+$SS
        $B=True
        $isArchived=False   # <<< FIXED: mark as not archived if we got a valid deeplink
      ENDIF

      # Fallback for Deeplink. Get some data for the Input
      IF Not($B)
        $advName=Sqlite("sql","SELECT AdvName FROM CW.FoundLabStages WHERE AdvId='$AdvId' LIMIT 1")
        MSGOK Msg="The Adventure '$advName', is not accessible from the API.$br This may work to automatically create the Adventure, otherwise, the macro will need your help to create the Adventure Labs.$p The macro will now open your Adventure Lab Finds.$br Try and locate the Adventure.$br Tip: Set the 'Adventures per page:' to 50 and select the previous  '<'   button to go to your oldest Finds, where Archives are more likely.$p     $advName$p Click on the Adventure Title.$br If it opens the Adventure page, Copy its URL.$br Return to the macro and paste the URL into the Input.$br Otherwise, if the Adventure page was not found, do not enter anything, as this will cause an error.$p Do not worry, the macro will help you create your Archived Adventure Lab Finds if this attempt fails."
        WEB Url="https://labs.geocaching.com/logs"

        INPUT Msg="Enter the WHOLE URL for the '$advName' Adventure (or leave blank if no Adventure page was found):" VarName=$input
        $url=$input

        IF RegEx("^https:\/\/labs\.geocaching\.com\/goto\/.+$",$url)
          $isArchived=False
          $B=True
        ELSE
          MSGOK Msg="Sorry, Adventure '$advName' is not available in the API or from your Adventure Lab Finds web page.$p We need to try another way to create this Adventure and its Labs."
          $isArchived=True
        ENDIF
      ENDIF
    ENDIF

    IF $isArchived
      GOSUB Name=CreateArchivedAdventure
    ELSE
      # API call was good or we got valid deeplink
      IF Not($B)
        $url=Sqlite("sql","SELECT deepLink FROM ApiMaster")
      ENDIF

      $deepLink=RegExSub("^https:\/\/labs\.geocaching\.com\/goto\/(.+)$",$url,1,1)
      GOSUB Name=AdvUTF8
    ENDIF

    SHOWSTATUS Display=Off
  ELSE
    MSGOK Msg="Enter an Adventure ID"
  ENDIF
ENDSUB

BEGINSUB Name=RepairAdventureLabs
# param: $advId
	$S="$p Repairing a corrupt Adventure: '$advId', that has an incorrect Guid for one or more of its Lab Stages.$p"
	GOSUB Name=ShowStatus

# Build a list of LabIds from FoundLabStages for this Adventure
  $codeList=Sqlite("sql","SELECT Group_Concat(Quote(LabId)) FROM CW.FoundLabStages WHERE AdvId='$advId'")

# Count how many of these LabIds exist in Caches
  $S=Sqlite("sql","SELECT Count(*) FROM Caches WHERE Guid IN ($codeList)")

# If mismatch (some Labs missing in Caches), fix them
  IF $S <> Sqlite("sql","SELECT Count(*) FROM CW.FoundLabStages WHERE AdvId='$advId'")
    
# Get coreCode for parent Adventure
    $coreCode=Sqlite("sql","SELECT SubStr(Code,3,4) FROM Caches WHERE Guid='$advId' LIMIT 1")

# Loop through all Lab stages in FoundLabStages
    $lab=1
    $labs=Val(Sqlite("sql","SELECT Count(*) FROM CW.FoundLabStages WHERE AdvId='$advId'"))
    WHILE $lab <= $labs

# Get update data
      $_Sql="SELECT LabId, LabName, AdvName FROM CW.FoundLabStages WHERE AdvId='$advId' ORDER BY LabFind LIMIT 1 OFFSET $lab-1"
			$data=Sqlite("sql", $_Sql)
      $labId=Extract($data,";",1)
      $labName=Extract($data,";",2)
      $advName=Extract($data,";",3)

# Construct Name and Code
			$name="$advName : $labName"
			$S=FormatNumber("00", $lab)
			$code="$LBprefix$coreCode$S"

# Get Lab Guid from Caches
      $guid=Sqlite("sql","SELECT Guid FROM Caches WHERE Code='$code'")

# Update Lab in Caches with the same Guid and Name as in your Lab Finds
      $S=Sqlite("sql","UPDATE Caches SET Guid='$labId', Name='$name' WHERE Code='$code'")

      $lab=$lab+1
    ENDWHILE
  ENDIF

# These labs now exist, so ensure they are updated not created
	$S=Sqlite("sql", "UPDATE CW.FoundLabStages SET Action='Update' WHERE AdvId='$advId' AND Action='Create'")
 debug status=on
ENDSUB

BEGINSUB Name=IDtoCoreCode62 # Parm: $cacheId Out: $coreCode
  $S=""
  $alphabet="0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz"
  $base=Len($alphabet)
  $length=4
  $idVal=Val($cacheId)
  WHILE $idVal > 0 
    $rem=$idVal Mod $base
    $S=SubStr($alphabet,$rem+1,1) + $S
    $idVal = Int($idVal / $base)
  ENDWHILE

  WHILE Len($S) < $length
    $S = "0" + $S
  ENDWHILE
  $coreCode=$S
ENDSUB
